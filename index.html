Montando un Sistema de Ficheros

sudo mount /dev/sda3 /mnt
for i in /dev /dev/pts /proc /sys; do sudo mount -B $i /mnt$i; done
sudo chroot /mnt
sudo grub-install /dev/sda3
sudo update-grub
exit
sudo for i in /sys /proc /dev/pts /dev; do sudo umount /mnt$i; done
